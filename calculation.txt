# Calculation: How many thousand hours would it take Kipchoge to run to the Moon?

# Given Data:
# - Kipchoge's marathon world record: 2:01:09 (Berlin 2022)
# - Marathon distance: 42.195 km
# - Moon minimum perigee (from Wikipedia): 356,400 km

# Step 1: Convert marathon time to hours
hours = 2
minutes = 1
seconds = 9
total_hours = hours + minutes/60 + seconds/3600
print(f"Kipchoge's world record time: {hours}:{minutes:02d}:{seconds:02d}")
print(f"Time in hours: {total_hours:.6f} hours")

# Step 2: Calculate marathon speed
marathon_km = 42.195
speed_km_per_hour = marathon_km / total_hours
print(f"\nMarathon distance: {marathon_km} km")
print(f"Speed: {speed_km_per_hour:.4f} km/h")

# Step 3: Calculate time to Moon
min_perigee_km = 356400
print(f"\nMoon minimum perigee: {min_perigee_km} km")
time_to_moon_hours = min_perigee_km / speed_km_per_hour
print(f"Time to reach Moon: {time_to_moon_hours:.2f} hours")

# Step 4: Round to nearest 1000 hours
rounded = round(time_to_moon_hours / 1000) * 1000
print(f"\nRounded to nearest 1000 hours: {int(rounded)} hours")
print(f"Answer: {int(rounded)}")
